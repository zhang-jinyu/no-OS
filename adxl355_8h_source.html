95 /*
96  * At +/- 2g with 20-bit resolution, scale is given in datasheet as
97  * 3.9ug/LSB = 0.0000039 * 9.80665 = 0.00003824593 m/s^2.
98  * For +/- 4g range a multiplier with value 2 is used.
99  * For +/-8g range, a multiplier with value 4 is used.
100  */
101 #define ADXL355_ACC_SCALE_FACTOR_MUL (int64_t) 38245
102 #define ADXL355_ACC_SCALE_FACTOR_DIV (int32_t)1000000000
103 
104 /*
105  * The datasheet defines an intercept of 1885 LSB at 25 degC
106  * and a slope of -9.05 LSB/C. The following formula can be used to find the
107  * temperature:
108  * Temp = ((RAW - 1885)/(-9.05)) + 25 but this doesn't follow the format of
109  * the IIO which is Temp = (RAW + OFFSET) * SCALE. Hence using some rearranging
110  * we get the scale as -0.110497238 and offset as -2111.25
111  */
112 /* OFFSET = ADXL355_TEMP_OFFSET / ADXL355_TEMP_OFFSET_DIV */
113 #define ADXL355_TEMP_OFFSET -211125
114 #define ADXL355_TEMP_OFFSET_DIV 100
115 
116 /* SCALE = ADXL355_TEMP_SCALE_FACTOR / ADXL355_TEMP_SCALE_FACTOR_DIV
117  We want to return the temperature in millidegrees Celsius so instead of
118  0.110497238, we are going to use 110.497238 value for scale
119  */
120 #define ADXL355_TEMP_SCALE_FACTOR -110497238
121 #define ADXL355_TEMP_SCALE_FACTOR_DIV 1000000
